POJ 1002 电话号码

2013-09-26

文章目录

Input

输入的格式是，第一行是一个正整数，指定电话号码薄中号码的数量（最多100000）。余下的每行是一个电话号码。每个电话号码由数字，大写字母（除了Q和Z）以及连接符组成。每个电话号码中只会刚好有7个数字或者字母。 Output

对于每个出现重复的号码产生一行输出，输出是号码的标准格式紧跟一个空格然后是它的重复次数。如果存在多个重复的号码，则按照号码的字典升序输出。如果输入数据中没有重复的号码，输出一行：

No duplicates.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define SIZE 10000000
int num[SIZE];

int trans(char x)
{
    if (x >= '0' && x <= '9') return x - '0';
    if (x == 'A' || x == 'B' || x == 'C') return 2;
    if (x == 'D' || x == 'E' || x == 'F') return 3;
    if (x == 'G' || x == 'H' || x == 'I') return 4;
    if (x == 'J' || x == 'K' || x == 'L') return 5;
    if (x == 'M' || x == 'N' || x == 'O') return 6;
    if (x == 'P' || x == 'R' || x == 'S') return 7;
    if (x == 'T' || x == 'U' || x == 'V') return 8;
    if (x == 'W' || x == 'X' || x == 'Y') return 9;
}

int main(int argc, char *argv[])
{
    int n;
    scanf("%d", &n);

    int i, j;
    char x[1000];
    for (i = 0; i < n; i++)
    {
        scanf("%s", x);
        int base =1, sum = 0;
        for (j = strlen(x) - 1; j >= 0; j--)
        {
            if (x[j] == '-') continue;
            sum += base * trans(x[j]);
            base = base * 10;
        }
        num[sum]++;
    }

    int p = 0;
    for (i = 0; i < SIZE; i++)
    {
        if (num[i] > 1)
        {
            p = 1;
            printf("%03d-%04d %d\n", i / 10000, i % 10000, num[i]);
        }
    }
    if (!p) printf("No duplicates.\n");
    return 0;
}
